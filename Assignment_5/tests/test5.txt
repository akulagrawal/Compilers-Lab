int sum(int a, int b) {
    return a + b;
}

int main() {
    int arr[4];
    arr[0] = 1;
    arr[1] = 2;
    arr[2] = 3;
    arr[3] = 4;

    int param0 = arr[0];
    int param1 = arr[1];
    int param2 = arr[2];
    int param3 = arr[3];
    int temp = sum(param0, param1);
    param0 = sum(temp, param2);
    param1 = sum(param0, param3);

    return param1;
}
