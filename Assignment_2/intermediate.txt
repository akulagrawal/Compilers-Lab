        class DefaultConstructor2 {
        int a;
        boolean b;
        private DefaultConstructor2() {
            a = 0;
            b = false;
        }
        private DefaultConstructor2(int c) {
            a = c;
            b = false;
        }
        public static void main(String[] args) {
            DefaultConstructor obj = new DefaultConstructor();

        }
